get up and running in 30 seconds

$ curl -fsSL https://raw.githubusercontent.com/jasondcamp/pingway/main/install.sh | bash

# then visit http://localhost:8080
$ docker run -d --name pingway \
--network host --cap-add NET_RAW \
-v pingway-data:/data \
--restart unless-stopped \
ghcr.io/jasondcamp/pingway:latest

# host networking + NET_RAW = true latency.
# macOS/Windows: use -p 8080:8080 instead.
